\section{Les 2: Verzamelingen, Relaties, Functies en Permutaties}

\subsection{Verzamelingen}

Een \inlinetext{verzameling} is een collectie van welbepaalde, onderling verschillende objecten, die we \inlinetext{elementen} noemen.

\SimpleHeading{Notatie}

We schrijven een verzameling met accolades \inlinecode{\{\dots\}}:
\begin{itemize}
    \item Door elementen op te sommen: $\{5, 6, 7\}$
    \item Door een regel: $\{2, 4, 6, 8, \dots\}$
    \item Door een eigenschap: $\{x \in \mathbb{Q} \mid 0 < x < 1\}$ (lees $\mid$ als "zodanig dat")
\end{itemize}

Het symbool \inlinetext{$\in$} betekent "is element van", en \inlinetext{$\notin$} is de ontkenning.

\textbf{Let op:} elementen in een verzameling zijn altijd verschillend. Dus $\{a, a\} = \{a\}$. Maar $\{a, \{a\}\}$ is een andere verzameling (twee elementen: $a$ en de verzameling $\{a\}$).

\SimpleHeading{Belangrijke Getallenverzamelingen}

\begin{itemize}
    \item $\mathbb{N} = \{0, 1, 2, 3, \dots\}$: de natuurlijke getallen, en $\mathbb{N}_0$: de natuurlijke getallen zonder $0$.
    \item $\mathbb{Z} = \{\dots, -2, -1, 0, 1, 2, \dots\}$: de gehele getallen.
    \item $\mathbb{Q}$: de rationale getallen.
    \item $\mathbb{R}$: de reële getallen.
    \item $\mathbb{C}$: de complexe getallen.
\end{itemize}

Deelverzamelingen worden aangeduid met boven- en benedenindexen: een bovenindex \inlinetext{$+$} betekent niet-negatief, \inlinetext{$-$} betekent niet-positief, en een benedenindex \inlinetext{$0$} sluit nul uit. \newline
Bijvoorbeeld: $\mathbb{Z}_0^{+} = \{1, 2, 3, \dots\}$ en $\mathbb{Q}^{-} = \{x \in \mathbb{Q} \mid x \leq 0\}$.

\SimpleHeading{Basisdefinities}

\begin{nicedefinition}*[Gelijkheid]
    \nextpart[Gelijkheid:]*
    $A = B$ als en slechts als ze dezelfde elementen hebben:
    \begin{equation*}
        A = B \quad \Leftrightarrow \quad (x \in A \Leftrightarrow x \in B)
    \end{equation*}
\end{nicedefinition}

\begin{nicedefinition}*[Deelverzameling]
    \nextpart[Deelverzameling:]*
    $A \subseteq B$ als en slechts als elke element van $A$ ook in $B$ zit.
    \begin{equation*}
        A \subseteq B \quad \Leftrightarrow \quad (x \in A \Rightarrow x \in B)
    \end{equation*}
\end{nicedefinition}

\begin{nicedefinition}*[Echte (strikte) deelverzameling]
    \nextpart[Echte (strikte) deelverzameling:]*
    $A$ is een echte (strikte) deelverzameling van $B$ als en slechts als $A$ een deelverzameling is van $B$ en niet gelijk aan $B$.
    \begin{equation*}
        A \subset B \quad \Leftrightarrow \quad A \subseteq B \;\;\&\;\; A \neq B
    \end{equation*}
\end{nicedefinition}

\begin{niceexample}*
    \nextpart[]*

    Zij $A = \{1, 2, 3\}$ en $B = \{1, 2, 3, 4\}$. Dan geldt $A \subseteq B$ en zelfs $A \subset B$. Want $4 \in B$, maar $4 \notin A$.
\end{niceexample}

\begin{niceexample}*
    \nextpart[]*

    $\{1, 2\} \subseteq \{1, 2\}$ is waar, maar $\{1, 2\} \subset \{1, 2\}$ is onwaar (ze zijn gelijk).
\end{niceexample}

\SimpleHeading{Bewerkingen op Verzamelingen}

\begin{nicedefinition}*[Unie (vereniging)]
    \nextpart[Unie:]*

    \begin{equation*}
        A \cup B : \{x \mid x \in A \lor x \in B\}
    \end{equation*}
\end{nicedefinition}

\begin{nicedefinition}*[Doorsnede]
    \nextpart[Doorsnede:]*

    \begin{equation*}
        A \cap B : \{x \mid x \in A \land x \in B\}
    \end{equation*}
\end{nicedefinition}

\newpage
\begin{nicedefinition}*[Nog wat extra verzamelingen]
    \nextpart[Extra verzamelingen:]*

    \begin{itemize}
        \item \textbf{Universele verzameling:} $U =$ de verzameling van elle mogelijke elementen in de context.
        \item \textbf{Ledige verzameling:} $\varnothing =$ de verzameling zonder elementen.
        \item Voor elke verzameling $A$ geldt: $\varnothing \subseteq A \text{ en } A \subseteq U$.
        \item \textbf{Singleton:} een verzameling met precies één element, bv. $\{\varnothing\}$ (bevat één element, namelijk de lege verzameling).
        \item \textbf{Disjunct:} $A$ en $B$ zijn disjunct $\Longleftrightarrow A \cap B = \varnothing$.
        \item \textbf{Complement:} $A' = \{x \mid x \notin A\}$ (alle elementen van $U$ die niet in $A$ zitten). Andere notaties die ook wel gebruikt worden zijn: $A^c$ of $\bar{A}$.
    \end{itemize}
\end{nicedefinition}

\begin{niceexample}*
    Zij $U = \{1,2,3,4,5\}$, $A = \{1, 2, 3\}$, $B = \{3,4,5\}$.
    \nextpart[Dan:]*
    \begin{itemize}
        \item $A \cup B = \{1,2,3,4,5\} = U$
        \item $A \cap B = \{3\}$
        \item $A' = \{4,5\}$
        \item $B' = \{1,2\}$
        \item $A$ en $B$ zijn niet disjunct (want 3 zit in beide)
    \end{itemize}
\end{niceexample}

\subsection{Algebra van de Verzamelingen}

Met de operatoren \inlinetext{$\cup$}, \inlinetext{$\cap$}, \inlinetext{$'$} en de relaties \inlinetext{$\subseteq$}, \inlinetext{$\subset$}, \inlinetext{$=$} kan men met verzamelingen "rekenen".

\SimpleHeading{Prioriteit van Operatoren}

De prioriteit van operatoren van hoog naar laag:
\begin{enumerate}
    \item $'$ (complement) $\rightarrow$ hoogste prioriteit
    \item $\cap$ (doorsnede)
    \item $\cup$ (unie)
    \item $\subset, \subseteq, =$ $\;\rightarrow$ laagste prioriteit
\end{enumerate}

Dus $A \cup B \cap C$ betekent $A \cup (B \cap C)$, net zoals bij $\times$ en $+$ in de gewone rekenkunde.

\SimpleHeading{Rekenregels}

\begin{nicetheorem}*[Rekenregels]
    Laat $A,B,C$ verzamelingen zijn. De belangrijkste regels:
    \nextpart[]*

    \begin{center}
        \begin{tabularx}{\textwidth}{@{} >{\centering\arraybackslash}p{3cm} X l @{}}
            \toprule
            \textbf{Nr} & \textbf{Regel} & \textbf{Naam} \\
            \midrule
            \multirow{2}{*}{5, 6}   & $A \cup (A \cap B) = A$                          & \multirow{2}{*}{Absorptie} \\
                                    & $A \cap (A \cup B) = A$                          & \\
            \midrule
            \multirow{2}{*}{7, 8}   & $(A \cup B)' = A' \cap B'$                       & \multirow{2}{*}{De Morgan} \\ 
                                    & $(A \cap B)' = A' \cup B'$                       & \\
            \midrule
            \multirow{2}{*}{9, 10}  & $A \cup B = B \cup A$                            & \multirow{2}{*}{Commutativiteit} \\
                                    & $A \cap B = B \cap A$                            & \\
            \midrule
            \multirow{2}{*}{11, 12} & $A \cup A = A$                                   & \multirow{2}{*}{Idempotentie} \\
                                    & $A \cap A = A$                                   & \\
            \midrule
            13                      & $A \cup (B \cup C) = (A \cup B) \cup C$          & Associativiteit \\
            \midrule
            \multirow{2}{*}{14, 15} & $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$ & \multirow{2}{*}{Distributiviteit} \\
                                    & $A \cup (B \cap C) = (A \cup B) \cap (A \cup C)$ & \\
            \midrule
            \multirow{2}{*}{16, 17} & $A \cup \varnothing = A$                         & \multirow{2}{*}{Identiteit} \\
                                    & $A \cap U = A$                                   & \\
            \midrule
            \multirow{2}{*}{18, 19} & $A \cup U = U$                                   & \multirow{2}{*}{Nulwet} \\
                                    & $A \cap \varnothing = \varnothing$               & \\
            \midrule
            \multirow{2}{*}{20, 21} & $A \cup A' = U$                                  & \multirow{2}{*}{Complementwet} \\
                                    & $A \cap A' = \varnothing$                        & \\
            \bottomrule
        \end{tabularx}
    \end{center}
\end{nicetheorem}

\begin{niceexample}*[De Morgan]
    Zij $U = \{1,2,3,4,5\}$, $A = \{1,2\}$, $B = \{2,3\}$. Dan:
    \nextpart[]*
    \begin{itemize}
        \item $A \cup B = \{1,2,3\}$, dus $(A \cup B)' = \{4,5\}$
        \item $A' = \{3,4,5\}$, $B' = \{4,5\}$, dus $A' \cap B' = \{4,5\}$ \emoji{check-mark-button}
    \end{itemize}
\end{niceexample}

\begin{niceexample}*[Distributiviteit]
    Zij $A = \{1,2\}$, $B = \{2,3\}$, $C = \{3,4\}$. Verifieer $A \cap (B \cup C) = (A \cap B) \cup (A \cap C)$:
    \nextpart[Oplossing:]*
    \begin{itemize}
        \item $B \cup C = \{2,3,4\}$, dus $A \cap (B \cup C) = \{2\}$
        \item $A \cap B = \{2\}$, $A \cap C = \varnothing$, dus $(A \cap B) \cup (A \cap C) = \{2\}$ \emoji{check-mark-button}
    \end{itemize}
\end{niceexample}

\SimpleHeading{Verschiloperatoren}

\begin{nicedefinition}*[Verschiloperatoren]
    \nextpart[Verschiloperatoren:]*
    \begin{itemize}
        \item \textbf{Verschil:}
        \begin{equation*}
            A \backslash B = \{x \mid x \in A \text{ en } x \notin B\} = A \cap B'
        \end{equation*} 
        \item \textbf{Symmetrisch verschil:}
        \begin{equation*}
            A \triangle B = (A \backslash B) \cup (B \backslash A)
        \end{equation*}
        $\longrightarrow$ De elementen die in precies één van de twee verzamelingen zitten.
    \end{itemize}
\end{nicedefinition}

\begin{niceexample}*
    Gegeven de verzamelingen $A = \{1,2,3\}$ en $B = \{2,3,4\}$.
    \nextpart[Dan:]*
    \begin{itemize}
        \item $A \backslash B = \{1\}$
        \item $B \backslash A = \{4\}$
        \item $A \triangle B = \{1,4\}$
    \end{itemize}
\end{niceexample}

\SimpleHeading{Machtsverzameling}

De verzameling van alle deelverzamelingen van $A$ wordt genoteerd als $2^{A}$:
\begin{equation}
    2^{A} = \{S \mid S \subseteq A\}
\end{equation}

\begin{niceexample}*[Machtsverzameling]
    Gegeven de verzameling $A = \{1,2\}$.
    \nextpart[Dan:]*
    \begin{equation*}
        2^{A} = \{\varnothing, \{1\}, \{2\}, \{1,2\}\}
    \end{equation*}
\end{niceexample}

\begin{barbox}[SeaGreen][gray!5]
    \textcolor{SeaGreen}{\textbf{Merk op:}} als $A$ $n$ elementen heeft, dan heeft $2^{A}$ precies $2^n$ elementen.
\end{barbox}

\subsection{Geordend paar en Cartesisch product}

Een \inlinetext{geordend paar} (koppel) is een paar elementen in een vaste volgorde: $(x,y)$.

Belangrijk: de volgorde doet ertoe!
\begin{equation*}
    (a,b) = (c,d) \Longleftrightarrow a=c \text{ en } b=d
\end{equation*}
Dus $(1,2) \neq (2,1)$, in tegenstelling tot verzamelingen waar $\{1,2\} = \{2,1\}$.

Het \inlinetext{Cartesisch product} van $A$ en $B$:
\begin{equation}
    A \times B = \{(x,y) \mid x \in A \text{ en } y \in B\}
\end{equation}

\begin{niceexample}*[Cartesisch product]
    Gegeven $A = \{1,2\}$ en $B = \{a,b,c\}$
    \nextpart[Dan]*
    \begin{equation*}
        A \times B = \{(1,a), (1,b), (1,c), (2,a), (2,b), (2,c)\}
    \end{equation*}
    Merk op dat $\#(A \times B) = \#A \cdot \#B = 2 \cdot 3 = 6$.
\end{niceexample}
$A\times A$ wordt ook genoteerd als \inlinetext{$A^2$}. Uitbreiding naar $n$ factoren: $A^n = A \times A \times \dots \times A$ ($n$ keer), met $n$-koppels $(a_1, a_2, \dots, a_n)$.

\begin{nicetheorem}*[De distrubitiviteit van $\times$ over $\cap$ en $\cup$:]
    \nextpart[]*
    \begin{itemize}
        \item $A \times (B \cap C) = (A \times B) \cap (A \times C)$
        \item $A \times (B \cup C) = (A \times B) \cup (A \times C)$
        \item $(A \times B) \cap (C \times D) = (A \cap C) \times (B \cap D)$
    \end{itemize}
\end{nicetheorem}

\subsection{Relaties}

Als $x \in A$ en $y \in B$ in een bepaald verband staan, spreken we van een \inlinetext{relatie} van $A$ naar $B$. Dit noteer je als:
\begin{equation}
    xRy \quad \text{ of } \quad (x, y) \in R
\end{equation}

De \inlinetext{graaf} van een relatie $R$ is de verzameling koppels $\{(x,y) | xRy\}$. Aangezien de graaf de relatie volledig bepaald (en omgekeerd), identificeren we een relatie met haar graaf:
\begin{equation}
    R \subseteq A \times B
\end{equation}

\begin{niceexample}*
    \nextpart[Voorbeeld:]
    $A = \{2,3,4\}$, $B=\{1,2,3,4\}$. De relatie "$x \leq y$" geeft:
    \begin{equation*}
        R = \{(2,2), (2,3), (2,4), (3,3), (3,4), (4,4)\}
    \end{equation*}

            \input{figures/L2/example_2_5_1.tex}
        }
    \end{center}
\end{niceexample}

\SimpleHeading{Inverse Relatie}

Zij $R \in A \times B$ een relatie en $x \in A$ en $y \in B$, dan is:
\begin{equation}
    R^{-1} = \{(x,y) \mid (y,x) \in R\}
\end{equation}
Dus als $R=$ "is kleiner dan", dan is $R^{-1}=$ "is groter dan".

\SimpleHeading{Samenstelling van Relaties}

\begin{nicedefinition}*[Samenstelling van Relaties]
    Zij $\text{R} \subseteq A \times B$ en $S \subseteq B \times C$
    \nextpart[Dan] is de samenstelling $S \circ R$ ("S na R") gedefinieerd als:
    \begin{equation*}
        S \circ R = \{(x,y)\; |\; (\exists z)\left((x,z) \in R \land (z,y) \in S\right)\}
    \end{equation*}
\end{nicedefinition}

\begin{niceexample}*
    Beschouw de volgende familie-relaties:
    \nextpart[]*

    \begin{equation*}
        \begin{aligned}
            &\text{als } &xRz = "x \text{ is de vader van } z" \\
            &\text{en } &zSy = "z \text{ is getrouwd met } y" \\
            &\text{dan is } &x(S \circ R)y = "x \text{ is de schoonvader van } y" \\
        \end{aligned}
    \end{equation*}
\end{niceexample}

\newpage
\begin{niceexample}*
    Zij $A = \{1,2,3\}$ en $\text{R}=\{(1,2), (2,3)\}$ en $S = \{(2,1), (3,2)\}$, bepaal $S \circ \text{R}$.
    \nextpart[Oplossing:]*
    We zoeken koppels $(x,y)$ waarvoor er een $z$ bestaat met:
    \begin{equation*}
        (x,z) \in R \text{ en } (z,y) \in S
    \end{equation*}
    Dit geldt voor:
    \begin{itemize}
        \item $(1,2) \in R$ en $(2,1) \in S \longrightarrow (1,1) \in S \circ R$
        \item $(2,3) \in R$ en $(3,2) \in S \longrightarrow (2,2) \in S \circ R$
    \end{itemize}
    Dus $S \circ R = \{(1,1), (2,2)\}$.
\end{niceexample}

\SimpleHeading{Belangrijke Eigenschappen}

Zij $R$, $S$ en $T$ relaties zodanig dat de hiernavolgende samenstellingen gedefinieerd zijn, dan geldt:
\begin{itemize}
    \item $(T \circ S) \circ R = T \circ (S \circ R)$ $\longrightarrow$ Associativiteit
    \item $(R^{-1})^{-1} = R$
    \item $(S \circ R)^{-1} = R^{-1} \circ S^{-1}$ $\longrightarrow$ \textcolor{SeaGreen}{\textbf{Let op}} de omgekeerde volgorde! Analoog aan kousen/shoenen:
    \begin{center}
        aankleden = schoenen $\circ$ kousen, uitkleden = $\text{kousen}^{-1} \circ \text{schoenen}^{-1}$
    \end{center}
\end{itemize}

\SimpleHeading{Domein en Bereik}

Zij $R \in A \times B$ een relatie en $x \in A$ en $y \in B$, dan is:
\begin{itemize}
    \item $\text{dom } R = \{x \; \mid \; \exists y \;:\; (x,y) \in R\}$ --- de "input" elementen
    \item $\text{Ran } R = \{y \;\mid\; \exists x \;:\; (x,y) \in R\}$ --- de "output" elementen
\end{itemize}

\begin{barbox}[SeaGreen][gray!5]
    \textcolor{SeaGreen}{\textbf{Handig:}} Dom $R$ = Ran $\text{R}^{-1}$ en Ran $R$ = Dom $R^{-1}$.
\end{barbox}

\subsection{Afbeeldingen en Functies}

Er is een hiërarchie:
\begin{equation*}
    \{\text{afbeeldingen}\} \subset \{\text{functies}\} \subset \{\text{relaties}\}
\end{equation*}

\begin{nicedefinition}*[Functie]
    \nextpart[Functie.] 
    Een \textbf{functie} $f: A \mapsto B$ is een \textbf{relatie} die \textbf{éénwaardig} is: elk element van Dom $f$ wordt naar precies één element afgebeeld.

    \vspace{0.5cm}

    $f$ is een functie van $A$ naar $B$: $f: A \mapsto B$ asa
    \begin{enumerate}
        \item $f \subseteq A \times B$ ($f$ is een relatie van $A$ naar $B$)
        \item $(x, y_1) \in f \; \land \; (x, y_2) \in f \;\Longrightarrow \; y_1 = y_2$ (éénwaardig)
    \end{enumerate}
\end{nicedefinition}

\begin{nicedefinition}*[Afbeelding]
    \nextpart[Afbeelding.] 
    Een \textbf{afbeelding} $f:A\mapsto B$ is een \textbf{functie} waarvoor Dom $f=A$ (de functie is overal op $A$ gedefinieerd).

    \vspace{0.5cm}

    $f$ is een afbeelding van $A$ naar $B$: $f:A \mapsto B$ asa
    \begin{enumerate}
        \item $f$ is een functie van $A$ naar $B$
        \item $\forall x \in A, \; \exists y \in B \text{ zodat } (x,y) \in f$ (dom $f=A$)
    \end{enumerate}
\end{nicedefinition}

\begin{niceexample}*
    Gegeven $f: \mathbb{R} \rightarrow \mathbb{R}: x \mapsto 1/x$, is dit een relatie/functie/afbeelding?
    \nextpart[Oplossing:]*
    $f: \mathbb{R} \rightarrow \mathbb{R}: x \mapsto 1/x$ is een functie maar geen afbeelding want het is niet gedefinieerd voor $x=0$. Daarentegen is bijvoorbeeld $g: \mathbb{R} \rightarrow \mathbb{R}: x \mapsto x^2$ wel een afbeelding.
\end{niceexample}

\begin{barbox}[OrangeRed][gray!5]
    \textcolor{OrangeRed}{\textbf{Opmerking:}} Vanwege éénwaardigheid shcrijf je meestal gewoon $y = f(x)$ i.p.v. $(x,y) \in f$.
\end{barbox}

\begin{nicedefinition}*[Vast punt]
    \nextpart[Vast punt.] 
    Zij $f: A \rightarrow A$ een functie. Dan noemt een element $x \in A$ een invariant element of \textbf{vast punt} van $f$ indien:
    \begin{equation}
        f(x) = x \qedtag
    \end{equation} 
\end{nicedefinition}

\begin{niceexample}*
    Wat zijn de vaste punten voor $f: \mathbb{R} \rightarrow \mathbb{R}: x \mapsto x^2$?
    \nextpart[Oplossing.]*
    De vaste punten zijn de oplossingen van $x^2=x$. Dus $x=0$ en $x=1$.
\end{niceexample}

\subsection{Injectie, Surjectie, Bijectie}

\SimpleHeading{Definities}

\begin{nicedefinition}*
    Zij $f: A \rightarrow B$ een afbeelding. Dan:
    \nextpart[]*

    \textcolor{PineGreen!75!black}{\textbf{Injectie (1-op-1):}} verschillende inputs geven verschillende outputs.
    \begin{equation}
        x_1 \neq x_2 \; \Rightarrow \; f(x_1) \neq f(x_2) \qedtag
    \end{equation}
    Of equivalent:
    \begin{equation}
        f(x_1) = f(x_2) \; \Rightarrow \; x_1 = x_2 \qedtag
    \end{equation}

    \vspace{0.5cm}

    \textcolor{PineGreen!75!black}{\textbf{Surjectie (op):}} elk element van $B$ wordt "geraakt".
    \begin{equation}
        \forall y \in B, \; \exists x \in A: f(x) = y \qquad
        \text{ oftewel } \qquad Ran \; f = B \qedtag
    \end{equation}

    \vspace{0.5cm}

    \textcolor{PineGreen!75!black}{\textbf{Bijectie:}} zowel injectie als surjectie (1-op-1 verband).
\end{nicedefinition}

\begin{niceexample}*
    Geef aan voor de volgende relaties als ze injectief, surjectief, bijectief, of geen van beiden zij.

    1) Zij $A = \{1,2,3\}$ en $B = \{a,b,c,d\}$. Ook geldt: $f(1)=a$, $f(2)=a$ en $f(3)=b$.
    \begin{itemize}
        \item[\rightarrow] Injectie, maar geen surjectie want niet elk element uit $B$ wordt geraakt ($d$).
    \end{itemize}

    \vspace{0.5cm}
    \hrule
    \vspace{0.5cm}

    2) Zij $A = \{1,2,3\}$ en $B = \{a,b\}$. Ook geldt: $f(1)=a$, $f(2)=a$ en $f(3)=b$.
    \begin{itemize}
        \item[\rightarrow] Surjectie, want elke element van $B$ wordt geraakt, maar geen injectie, want $f(1) = f(2)$.
    \end{itemize}

    \vspace{0.5cm}
    \hrule
    \vspace{0.5cm}

    3) Zij $A = \{1,2,3\}$ en $B = \{a,b,c\}$. Ook geldt: $f(1)=b$, $f(2)=c$ en $f(3)=a$.
    \begin{itemize}
        \item[\rightarrow] Bijectie
    \end{itemize}
\end{niceexample}

\SimpleHeading{Bijzondere Afbeeldingen}

\begin{nicedefinition}*[Bijzondere Afbeeldingen]
    \nextpart[Enkele voorbeelden van (bijzondere) afbeeldingen:]*

    De \textcolor{PineGreen!75!black}{\textbf{identieke transformatie}} (van een verzameling $A$) is gedefinieerd als
    \begin{equation}
        1_A: A \rightarrow A: x \mapsto x, \quad \forall x \in A \qedtag
    \end{equation}
    Merk op dat $1_A$ altijd een bijectie is!

    \vspace{0.25cm}
    \hrule

    \vspace{0.5cm}

    Een \textcolor{PineGreen!75!black}{\textbf{constante afbeelding}} is gedefinieerd als
    \begin{equation}
        k_b: A \rightarrow B: x \mapsto b \in B, \quad \forall x \in A \qedtag
    \end{equation}
    Als $A$ maar 1 elemente heeft is $k_b$ een injectie. \newline
    Als $B$ maar 1 element heeft is $k_b$ een surjectie.

    \vspace{0.25cm}
    \hrule

    \vspace{0.5cm}

    De \textcolor{PineGreen!75!black}{\textbf{inlassing}} (van $A$ in $B$) is gedefinieerd indien $A \subseteq B$ als
    \begin{equation}
        i_A: A \rightarrow B: x \mapsto x, \quad \forall x \in A \subseteq B \qedtag
    \end{equation}
    Als $A=B$, dan is $i_A = 1_A$.
    \begin{center}
        \resizebox{0.5\textwidth}{!}{
            \input{figures/L2/inlassing_vis.tex}
        }
    \end{center}

    \vspace{0.25cm}
    \hrule

    \vspace{0.5cm}

    De \textcolor{PineGreen!75!black}{\textbf{karakteristieke afbeelding}} van $A$ in $C$ is gedefinieerd voor $A \subseteq C$ en $B = \{0,1\}$ als
    \begin{equation}
        E_A : C \rightarrow B : \begin{cases}
            x \mapsto 1 & \quad \text{ als } x \in A \\
            x \mapsto 0 & \quad \text{ als } x \notin A
        \end{cases} \qedtag
    \end{equation}
    Dus $E_A(A) = \{1\}$ en $E_A(C \backslash A) = \{0\}$.

    \vspace{0.25cm}
    \hrule

    \vspace{0.5cm}

    De \textcolor{PineGreen!75!black}{\textbf{Beperking}} van $f$ tot $C$ is gedefinieerd voor $f : A \rightarrow B$ en $C \subseteq A$ als
    \begin{equation}
        f_C: C \rightarrow B: x \mapsto f(x), \quad \forall x \in C \qedtag
    \end{equation}
    Dus $f_C = f \circ i_C$. Een functie beperkt tot haar domein wordt een afbeelding.
\end{nicedefinition}

\SimpleHeading{Samenstelling}

\begin{nicetheorem}*[Samenstelling]
    \nextpart[Samenstelling.]*
    Als $f:A \rightarrow B$ en $g: B \rightarrow C$ afbeeldingen zijn, dan is $g \circ f : A \rightarrow C$ ook een afbeelding met
    \begin{equation}
        (g \circ f)(x) = g(f(x)) \qedtag
    \end{equation}
\end{nicetheorem}

\begin{barbox}[OrangeRed][gray!5]
    \textcolor{OrangeRed}{\textbf{Belangrijk:}} samenstelling bewaart injectiviteit, surjectiviteir en bijectiviteit (als $f$ en $g$ beide injecties zijn, dan is $g \circ f$ ook een injectie, enz.)
\end{barbox}

\SimpleHeading{Inverse Afbeelding}

\begin{nicedefinition}*[Inverse Afbeelding]
    \nextpart[Inverse Afbeelding.]*
    Een functie $f : A \rightarrow B$ is \textbf{inverteerbaar} $\Longleftrightarrow f^{-1} : B \rightarrow A$ ook en functie is. $f^{-1}$ noemen we dan de inverse functie. Een afbeelding $f: A \rightarrow B$ is inverteerbaar $\Longleftrightarrow f^{-1} : B \rightarrow A$ ook een afbeelding is. $f^{-1}$ noemen we dan de inverse afbeelding van $f$.

    \vspace{0.5cm}

    Dit betekent dat voor een inverteerbare afbeelding $y = f(x) \Leftrightarrow f^{-1}(y)$, en dit $\forall x \in A$ en $\forall y \in B$. 

    \vspace{0.5cm}

    Een afbeelding $f : A \rightarrow B$ is \textbf{inverteerbaar} $\Longleftrightarrow f$ een \textbf{bijectie} is en de inverse afbeelding is ook een bijectie.
\end{nicedefinition}

\begin{nicetheorem}*
    Zij $f : A \rightarrow B$ en $g : B \rightarrow A$ twee afbeeldingen,
    \nextpart[Dan is]*
    \begin{equation}
        g = f^{-1} \Leftrightarrow g \circ f = 1_A \land f \circ g = 1_B \qedtag
    \end{equation}
\end{nicetheorem}

\begin{barbox}[OrangeRed][gray!5]
    \textcolor{OrangeRed}{\textbf{Let op:}} Als slechts aan één van de twee gelijkheden in het rechterlid van de vorige equivalentie voldaan is, zal $g$ niet noodzakelijk de inverse van $f$ zijn. Men spreekt dan van linker- en rechter inverse. 
\end{barbox}

\begin{nicedefinition}*[Linker- en Rechter Inverse]
    \nextpart[Definitie.]*
    Zij $f : A \rightarrow B$ een afbeelding. Een afbeelding $g : B \rightarrow A$ die voldoet aan
    \begin{equation*}
        g \circ f = 1_A
    \end{equation*}
    noemt men een \textcolor{PineGreen!75!black}{\textbf{linker inverse}} van $f$, en als ze voldoet aan
    \begin{equation*}
        f \circ g = 1_B
    \end{equation*}
    noemt men het een \textcolor{PineGreen!75!black}{\textbf{rechter inverse}} van $f$.
\end{nicedefinition}

\SimpleHeading{Permutaties}

Een \inlinetext{permutatie} is een bijectie van een eindige verzameling naar zichzelf. We werken met indices $\{1,2,\dots,n\}$ en noteren:
\begin{equation}
    \pi = \begin{pmatrix}
        1       & 2       & \cdots & n \\
        \pi (1) & \pi (2) & \cdots & \pi (n)
    \end{pmatrix}
\end{equation}

\textbf{Voorbeeld:} De permutatie $\pi = \left(1 \rightarrow 3, 2 \rightarrow 1, 3 \rightarrow 2, 4 \rightarrow 5, 5 \rightarrow 6, 6 \rightarrow 4\right)$ (zie \Cref{fig:l2_permutatie_en_cycli}):
\begin{equation*}
    \pi = \begin{pmatrix}
        1 & 2 & 3 & 4 & 5 & 6 \\
        3 & 1 & 2 & 5 & 6 & 4
    \end{pmatrix}
\end{equation*}

    \caption{Permutatie en cycli}
    \label{fig:l2_permutatie_en_cycli}
\end{figure}

\SimpleHeading{Cycli}

Een \inlinetext{cyclus} $(a_1 \; a_2 \; \dots \; a_r)$ verschuift de elementen cysclisch naar rechts: $a_1 \rightarrow a2$, \dots, $a_r \rightarrow a_1$ en laat alle andere elementen staan.

Een \inlinetext{transpositie} is een 2-cyclus: verwisseling van 2 elementen.

\SimpleHeading{Belangrijke Eigenschappen}

\begin{nicetheorem}*
    \nextpart[Belangrijke Eigenschappen:]*
    \begin{enumerate}
        \item Elke permutatie is te schrijven als een samenstelling van \textbf{disjuncte cycli}.
        \item Elke cyclus (en dus elke permutatie) is te schrijven als een samenstelling van \textbf{transposities}.
        \begin{equation}
            (a_1, a_2, \dots, a_k) = (a_1, a_k) \circ (a_1, a_{k-1}) \circ \dots \circ (a_1, a_2) \qedtag
        \end{equation}
    \end{enumerate}
\end{nicetheorem}

\begin{niceexample}*
    Neem terug de permutatie van eerder:
    \nextpart[]*
    \begin{equation*}
        \pi = (1 \; 3 \; 2) \circ (4 \; 5 \; 6)
    \end{equation*}
    Deze kunnen we volgens bovenstaande eigenschap verder ontbinden:
    \begin{itemize}
        \item $(1 \; 3 \; 2) = (1 \; 2) \circ (1 \; 3)$ en 
        \item $(4 \; 5 \; 6) = (4 \; 6) \circ (4 \; 5)$
    \end{itemize}
    $\longrightarrow$ Dus $\pi = (1 \; 2) \circ (1 \; 3) \circ (4 \; 6) \circ (4 \; 5)$. 
\end{niceexample}

\SimpleHeading{Teken van een Permutatie}

Het \inlinetext{aantal inversies} $I(\pi)$: tel voor elke positie $i$ het aantal posities $j > i$ waarvoor $\pi(j) < \pi(i)$.
\begin{itemize}
    \item $I(\pi)$ even $\rightarrow \pi$ is een \inlinetext{even} permutatie, $\text{sign}(\pi) = +1$
    \item $I(\pi)$ oneven $\rightarrow \pi$ is een \inlinetext{oneven} permutatie, $\text{sign}(\pi) = -1$
\end{itemize}

\begin{niceexample}*
    \nextpart[Voorbeeld:]*
    Beschouw opnieuw de permutatie:
    \begin{equation*}
        \pi = \begin{pmatrix}
            1 & 2 & 3 & 4 & 5 & 6 \\
            3 & 1 & 2 & 5 & 6 & 4
        \end{pmatrix}
    \end{equation*}
    Wat is het teken van deze permutatie?

    \vspace{0.5cm}

    \textbf{Oplossing:} Beschouw de rij $3, 1, 2, 5, 6, 4$, we tellen het aantal inversies $I(\pi)$:
    \begin{itemize}
        \item Na 3: 1 en 2 zijn kleiner dan 3 $\rightarrow$ 2
        \item Na 1: geen kleiner dan 1 $\rightarrow$ 0
        \item Na 2: geen kleiner dan 2 $\rightarrow$ 0
        \item Na 5: enkel 4 kleiner dan 5 $\rightarrow$ 1
        \item Na 6: 4 kleiner dan 6 $\rightarrow$ 1
        \item Na 4: geen getallen meer $\rightarrow$ 0
    \end{itemize}
    $\longrightarrow$ $I(\pi) = 2 + 0 + 0 + 1 + 1 + 0 = 4$, het is dus een \textbf{even} permutatie. 
\end{niceexample}

\SimpleHeading{Rekenregels}

\begin{nicetheorem}*
    \nextpart[Rekenregels:]*
    \begin{itemize}
        \item $\text{sign}(\pi \circ \rho) = \text{sign}(\pi) \circ \text{sign}(\rho)$
        \item Een transpositie is \textbf{altijd oneven}
        \item Een $r$-cyclus is \underline{even} $\Leftrightarrow$ $r$ is \underline{oneven} (en omgekeerd)
    \end{itemize}
\end{nicetheorem}
