% Factory: #1=base env (lemma), #2=proof env (lemproof), #3=wrapper (nicelemma), #4=color (purple)
\NewDocumentCommand{\createResEnviornment}{mmmm}{%
  % Numbered base theorem (resets each section)
  \newtheorem{#1}{\sffamily\bfseries\color{#4!50!black}\MakeUppercase{#1}}[section]
  % Unnumbered base theorem
  \newtheorem*{#1*}{\sffamily\bfseries\color{#4!50!black}\MakeUppercase{#1}}

  % Proof environment with optional title (now bold + colored)
  \NewDocumentEnvironment{#2}{ o }{%
    \IfNoValueTF{##1}{%
      \renewcommand{\proofname}{\textbf{\color{#4!50!black}Proof}}%
      \begin{proof}%
    }{%
      \begingroup
      \renewcommand{\proofname}{\textbf{\color{#4!50!black}##1}}%
      \begin{proof}%
    }%
    % Define the symbol but do NOT auto-place it:
    \renewcommand{\qedsymbol}{\textcolor{#4!40!black}{\rule{1.5ex}{1.5ex}}}%
  }{%
  \end{proof}%
  \IfNoValueTF{##1}{}{%
    \endgroup
  }%
}

  % ---------------- Numbered wrapper ----------------
  \NewDocumentEnvironment{#3}{s o}{%
    \IfBooleanTF{##1}{%
      \begin{tcolorbox}[
        enhanced,colback=#4!8,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable,after skip=0pt]%
    }{%
      \begin{tcolorbox}[
        enhanced,colback=#4!8,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable]%
    }%
    \IfNoValueTF{##2}{\begin{#1}}{\begin{#1}[##2]}%
    % Local helper to split statement/proof
    % --- FIX 2: Replaced \newcommand with \NewDocumentCommand ---
    \NewDocumentCommand{\nextpart}{ o s }{%
      \end{#1}%
      \end{tcolorbox}%
      \begin{tcolorbox}[
        enhanced,colback=#4!2,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable,before skip=0pt]%
      % Check if \nextpart was given an argument (####1)
      \IfNoValueTF{####1}{%
          \begin{#2}% If not, call proof env with no argument
      }{%
          \begin{#2}[####1]% If so, pass argument to proof env
      }%
      \IfBooleanTF{####2}{\par\noindent}{}%
    }%
  }{%
    \IfBooleanTF{##1}{\end{#2}}{\end{#1}}%
    \end{tcolorbox}%
  }%

  % ---------------- Unnumbered wrapper ----------------
  \NewDocumentEnvironment{#3*}{s o}{%
    \IfBooleanTF{##1}{%
      \begin{tcolorbox}[
        enhanced,colback=#4!8,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable,after skip=0pt]%
    }{%
      \begin{tcolorbox}[
        enhanced,colback=#4!8,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable]%
    }%
    \IfNoValueTF{##2}{\begin{#1*}}{\begin{#1*}[##2]}%
    % Local helper to split statement/proof
    % --- FIX 2: Replaced \newcommand with \NewDocumentCommand ---
    \NewDocumentCommand{\nextpart}{ o s }{%
      \end{#1*}%
      \end{tcolorbox}%
      \begin{tcolorbox}[
        enhanced,colback=#4!2,fonttitle=\bfseries,sharp corners,
        boxrule=0pt,frame hidden,borderline west={2pt}{0pt}{#4!70},
        breakable,before skip=0pt]%
      % Check if \nextpart was given an argument (####1)
      \IfNoValueTF{####1}{%
          \begin{#2}% If not, call proof env with no argument
      }{%
          \begin{#2}[####1]% If so, pass argument to proof env
      }%
      \IfBooleanTF{####2}{\par\noindent}{}%
    }%
  }{%
    \IfBooleanTF{##1}{\end{#2}}{\end{#1*}}%
    \end{tcolorbox}%
  }%
}

% ---------- Code for Equation Boxes ------------
\NewDocumentEnvironment{equationbox}{ s O{black} O{} +b }
{%
    % 1. Helper: Define the transparency/fill style safely
    %    (Solves the "white not defined" and opacity issues)
    \if\relax\detokenize{#3}\relax
        % No fill color -> Transparent
        \tcbset{eqfill/.style={opacityback=0, colback=white}}
    \else
        % Fill color provided -> Solid
        \tcbset{eqfill/.style={opacityback=1, colback=#3}}
    \fi
    
    % 2. Define the specific box appearance
    \tcbset{eqbox/.style={
        eqfill,                % Apply the fill logic calculated above
        sharp corners,
        boxrule=0.8pt,
        colframe=#2,           % Line color from argument
        % Dimensions (Tight fit)
        top=3pt, bottom=3pt, left=5pt, right=5pt, boxsep=1pt,
        % Alignment properties
        nobeforeafter,         % Ensures it behaves like a word, not a paragraph
        tcbox raise base,      % Aligns the math baseline correctly
        enhanced               % Required for 'standard jigsaw' transparency
    }}

    % 3. Output the Environment
    %    We wrap a \tcbox INSIDE standard math environments.
    \IfBooleanTF{#1}
    {% Starred version: Unnumbered \[ ... \]
        \[
            \tcbox[eqbox, standard jigsaw]{\ensuremath{\displaystyle #4}}
        \]
    }
    {% Standard version: Numbered \begin{equation} ...
        \begin{equation}
            \tcbox[eqbox, standard jigsaw]{\ensuremath{\displaystyle #4}}
        \end{equation}
    }
}{}

% barbox environment: \begin{barbox}[<barcolor>][<bgcolor>] ... \end{barbox}
\NewDocumentEnvironment{barbox}{ O{gray} o }{%
  % #1 = bar color  (optional, default gray)
  % #2 = background (optional, if absent use white)
  \IfNoValueTF{#2}{%
    % No background color given -> use white (looks transparent on white paper)
    \begin{tcolorbox}[
      enhanced,
      breakable,
      colback=white,
      colframe=white,
      borderline west={3pt}{0pt}{#1},
      left=10pt,
      right=0pt,
      boxrule=0pt,
    ]%
  }{%
    % Background color given -> use it
    \begin{tcolorbox}[
      enhanced,
      breakable,
      colback=#2,
      colframe=white,
      borderline west={3pt}{0pt}{#1},
      left=10pt,
      right=0pt,
      boxrule=0pt,
    ]%
  }%
}{%
  \end{tcolorbox}%
}
